pipeline {
    agent any

    tools {
        jdk 'jdk17'
        nodejs 'node23'
    }

    stages {

        stage('Git Checkout') {
            steps {
                git 'https://github.com/adgudewar/DevOps-Project-Zomato-clone-deployment.git'
            }
        }

        stage('File System Scan') {
            steps {
                sh 'trivy fs . > trivy.txt'
            }
        }

        stage('SonarQube Scanner') {
            steps {
                script {
                    def scannerHome = tool 'sonar-scanner'

                    withSonarQubeEnv('sonar-scanner') {
                        sh """
                            ${scannerHome}/bin/sonar-scanner \
                            -Dsonar.projectName=zomato \
                            -Dsonar.projectKey=zomato
                        """
                    }
                }
            }
        }

        stage('Code Quality Gate') {
            steps {
                script {
                    waitForQualityGate abortPipeline: false
                }
            }
        }

        stage('Application Build') {
            steps {
                sh 'npm install'
                sh 'npm run build'
            }
        }

        stage('Application Image Build') {
            steps {
                sh 'docker build -t zomato .'
            }
        }

        stage('Tag & Push to Docker Registry') {
            steps {
                withDockerRegistry(
                    credentialsId: 'docker',
                    url: 'https://index.docker.io/v1/'
                ) {
                    sh 'docker tag zomato anand1g/zomato:latest'
                    sh 'docker push anand1g/zomato:latest'
                }
            }
        }

        stage('Deploy to Container') {
            steps {
                sh '''
                    docker rm -f zomato || true
                    docker run -d --name zomato -p 3000:3000 anand1g/zomato:latest
                '''
            }
        }
    }

    post {
        always {
            emailext(
                attachLog: true,
                subject: "${currentBuild.currentResult}: ${env.JOB_NAME}",
                body: """
                <html>
                <body>
                    <h3>Jenkins Build Notification</h3>
                    <p><b>Project:</b> ${env.JOB_NAME}</p>
                    <p><b>Build Number:</b> ${env.BUILD_NUMBER}</p>
                    <p><b>Status:</b> ${currentBuild.currentResult}</p>
                    <p><b>Build URL:</b> ${env.BUILD_URL}</p>
                </body>
                </html>
                """,
                to: 'anandgudewar857@gmail.com',
                mimeType: 'text/html',
                attachmentsPattern: 'trivy.txt'
            )
        }
    }
}
